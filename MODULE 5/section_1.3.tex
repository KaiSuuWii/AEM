\subsection*{1.3 Differentiation of Vector Functions}

\begin{conceptbox}
    The derivative is taken component-wise:
    \begin{equation*}
        \mathbf{r}'(t) = \langle x'(t), y'(t), z'(t) \rangle
    \end{equation*}
\end{conceptbox}

\begin{examplebox}
    Differentiate $\mathbf{r}(t)=\langle t^2, \sin t, e^t\rangle$.
    \newline \textbf{Solution:}
    \begin{align*}
        \mathbf{r}'(t)=\langle 2t, \cos t, e^t\rangle
    \end{align*}
\end{examplebox}

\begin{examplebox}
    Find velocity and acceleration for $\mathbf{r}(t)=\langle t, t^2, t^3\rangle$.
    \newline \textbf{Solution:}
    \begin{align*}
        \mathbf{v}(t) & = \langle1,2t,3t^2\rangle \\
        \mathbf{a}(t) & = \langle0,2,6t\rangle
    \end{align*}
\end{examplebox}

\begin{examplebox}
    Find tangent vector at $t=1$ for $\mathbf{r}(t)=\langle \ln t, t^2, e^t\rangle$.
    \newline \textbf{Solution:}
    \begin{align*}
        \mathbf{r}'(t) & = \left\langle \frac{1}{t}, 2t, e^t \right\rangle \\
        \mathbf{r}'(1) & = \langle 1, 2, e \rangle
    \end{align*}
\end{examplebox}
