\section*{Activity: Vector Calculus and Integral Theorems}

Answer the following questions clearly and show complete solutions where necessary.

\begin{enumerate}
    \item Let
    \[
    f(x,y)=x^2+xy+y^2.
    \]
    Find the directional derivative of $f$ at the point $(1,2)$ in the direction of the vector $\langle 3,4\rangle$.

    \item Evaluate the line integral
    \[
    \int_C \mathbf{F}\cdot d\mathbf{r}
    \]
    where
    \[
    \mathbf{F}(x,y)=\langle y,\; x\rangle
    \]
    and
    \[
    \mathbf{r}(t)=\langle t,\; t^2\rangle,\qquad 0\le t\le 1.
    \]

    \item Use Green's Theorem to evaluate
    \[
    \oint_C (x^2y\,dx+xy^2\,dy)
    \]
    where $C$ is the positively oriented boundary of the triangle with vertices $(0,0)$, $(1,0)$, and $(1,2)$.

    \item Use Stokes' Theorem to evaluate
    \[
    \oint_C \mathbf{F}\cdot d\mathbf{r}
    \]
    where
    \[
    \mathbf{F}(x,y,z)=\langle -y,\; x,\; z\rangle
    \]
    and $C$ is the boundary of the disk
    \[
    x^2+y^2\le 1
    \]
    in the plane $z=1$.

    \item Use the Divergence Theorem to compute the outward flux of
    \[
    \mathbf{F}(x,y,z)=\langle x^3,\; y^3,\; z^3\rangle
    \]
    across the surface of the unit sphere
    \[
    x^2+y^2+z^2=1.
    \]
\end{enumerate}
