\section*{4 Curl and Divergence}

\subsection*{4.1 Vector Fields}

\begin{conceptbox}
    A vector field assigns a vector to every point in space:
    \begin{equation*}
        \mathbf{F}(x,y,z) = \langle P(x,y,z), Q(x,y,z), R(x,y,z) \rangle
    \end{equation*}
\end{conceptbox}

Vector fields are used to model physical quantities such as velocity of a fluid, electric fields, and force fields.

\begin{examplebox}
    Determine the value of the vector field $\mathbf{F}(x,y,z)=\langle x, y, z \rangle$ at $(1,2,3)$.

    \begin{align*}
        \mathbf{F}(1,2,3) = \langle 1,2,3 \rangle
    \end{align*}

    \textbf{Interpretation:} At the point $(1,2,3)$, the vector points outward from the origin.
\end{examplebox}
