\subsection*{1.5 Arc Length and Speed}

\begin{conceptbox}
    Speed is magnitude of velocity:
    \begin{equation*}
        |\mathbf{r}'(t)|
    \end{equation*}
    Arc length:
    \begin{equation*}
        L = \int_a^b |\mathbf{r}'(t)| dt
    \end{equation*}
\end{conceptbox}

\begin{examplebox}
    Find speed for $\mathbf{r}(t)=\langle \cos t, \sin t, t\rangle$.
    \newline \textbf{Solution:}
    \begin{align*}
        \mathbf{r}'(t)   & = \langle -\sin t, \cos t, 1 \rangle \\
        |\mathbf{r}'(t)| & = \sqrt{\sin^2 t + \cos^2 t + 1}     \\
                         & = \sqrt{2}
    \end{align*}
\end{examplebox}

\begin{examplebox}
    Find arc length from $0$ to $1$.
    \newline \textbf{Solution:}
    \begin{align*}
        L & = \int_0^1 \sqrt{2}\, dt \\
          & = \sqrt{2}
    \end{align*}
\end{examplebox}
