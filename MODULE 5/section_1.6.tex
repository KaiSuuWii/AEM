\subsection*{1.6 Unit Tangent, Normal, and Curvature}

\begin{conceptbox}
    Unit tangent:
    \begin{equation*}
        \mathbf{T} = \frac{\mathbf{r}'(t)}{|\mathbf{r}'(t)|}
    \end{equation*}
\end{conceptbox}

\begin{examplebox}
    Find unit tangent for $\mathbf{r}(t)=\langle t, t^2, 0\rangle$.
    \newline \textbf{Solution:}
    \begin{align*}
        \mathbf{r}'(t)   & = \langle 1, 2t, 0 \rangle                         \\
        |\mathbf{r}'(t)| & = \sqrt{1 + 4t^2}                                  \\
        \mathbf{T}(t)    & = \frac{\langle 1, 2t, 0 \rangle}{\sqrt{1 + 4t^2}}
    \end{align*}
\end{examplebox}
