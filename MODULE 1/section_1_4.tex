\subsection*{1.4 Taylor Series}

If a function $f(x)$ has derivatives of all orders at $x=a$, then its Taylor series is
\[
    f(x) = \sum_{n=0}^{\infty} \frac{f^{(n)}(a)}{n!}(x-a)^n
\]

\paragraph{Example 1: Taylor Series of $e^x$ about $a=1$}
\[
    e^x = e \left[ 1 + (x-1) + \frac{(x-1)^2}{2!} + \cdots \right]
\]

\paragraph{Example 2: Taylor Series of $\ln x$ about $a=1$}
\[
    \ln x = (x-1) - \frac{(x-1)^2}{2} + \frac{(x-1)^3}{3} - \cdots
\]

\paragraph{Example 3: Polynomial Approximation}
Approximate $\sqrt{x}$ near $x=4$:
\[
    \sqrt{x} = 2 + \frac{1}{4}(x-4) - \frac{1}{64}(x-4)^2 + \cdots
\]
