\subsection{Subspaces}

A subspace is a smaller vector space inside a larger one. It is very common in linear algebra and engineering analysis.

\begin{conceptbox}
    A subset $W$ of a vector space $V$ is called a \textbf{subspace} of $V$ if:
    \begin{enumerate}[label=(\alph*)]
        \item $\mathbf{0}\in W$
        \item If $\mathbf{u},\mathbf{v}\in W$, then $\mathbf{u}+\mathbf{v}\in W$
        \item If $\mathbf{u}\in W$ and $c$ is a scalar, then $c\mathbf{u}\in W$
    \end{enumerate}
\end{conceptbox}

These conditions are often called the \textbf{subspace test}.

\begin{examplebox}
    Determine if the set
    \[
        W=\{[x,y]\in\mathbb{R}^2 \mid y=2x\}
    \]
    is a subspace of $\mathbb{R}^2$.

    \textbf{Step 1: Check if the zero vector is included.}

    If $x=0$, then $y=2(0)=0$, so $[0,0]\in W$.

    \textbf{Step 2: Closure under addition.}

    Let $\mathbf{u}=[x_1,2x_1]$ and $\mathbf{v}=[x_2,2x_2]$ be in $W$.

    Then
    \[
        \mathbf{u}+\mathbf{v}=[x_1+x_2,2x_1+2x_2]
        =[x_1+x_2,2(x_1+x_2)].
    \]

    This satisfies the condition $y=2x$, so $\mathbf{u}+\mathbf{v}\in W$.

    \textbf{Step 3: Closure under scalar multiplication.}

    Let $c$ be a scalar. Then:
    \[
        c\mathbf{u}=c[x_1,2x_1]=[cx_1,2(cx_1)].
    \]

    This still satisfies $y=2x$. Thus $c\mathbf{u}\in W$.

    Therefore, $W$ is a subspace of $\mathbb{R}^2$.
\end{examplebox}

\begin{examplebox}
    Determine if the set
    \[
        S=\{[x,y]\in\mathbb{R}^2 \mid x+y=1\}
    \]
    is a subspace.

    The zero vector $[0,0]$ does not satisfy $0+0=1$.
    So $[0,0]\notin S$.

    Thus, $S$ is not a subspace.
\end{examplebox}
