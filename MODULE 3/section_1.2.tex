\subsection{Basic Vector Operations}

Let $\mathbf{u}, \mathbf{v} \in \mathbb{R}^n$ and let $c$ be a scalar.

\subsubsection{Vector Addition}

\begin{conceptbox}
    If
    \[
        \mathbf{u} = [u_1,u_2,\dots,u_n], \qquad
        \mathbf{v} = [v_1,v_2,\dots,v_n],
    \]
    then
    \[
        \mathbf{u}+\mathbf{v} = [u_1+v_1, u_2+v_2, \dots, u_n+v_n].
    \]
\end{conceptbox}

\subsubsection{Vector Subtraction}

\[
    \mathbf{u}-\mathbf{v} = [u_1-v_1, u_2-v_2, \dots, u_n-v_n].
\]

\subsubsection{Scalar Multiplication}

\begin{conceptbox}
    If $c$ is a scalar and $\mathbf{v}=[v_1,v_2,\dots,v_n]$, then
    \[
        c\mathbf{v} = [cv_1, cv_2, \dots, cv_n].
    \]
\end{conceptbox}

\subsubsection{Basic Properties of Vector Operations}

For vectors $\mathbf{u}, \mathbf{v}, \mathbf{w} \in \mathbb{R}^n$ and scalar $c$, the following hold:

\begin{itemize}
    \item Commutativity: $\mathbf{u}+\mathbf{v} = \mathbf{v}+\mathbf{u}$
    \item Associativity: $(\mathbf{u}+\mathbf{v})+\mathbf{w} = \mathbf{u}+(\mathbf{v}+\mathbf{w})$
    \item Distributive Law: $c(\mathbf{u}+\mathbf{v}) = c\mathbf{u}+c\mathbf{v}$
    \item Distributive Law: $(c+d)\mathbf{v} = c\mathbf{v}+d\mathbf{v}$
\end{itemize}
