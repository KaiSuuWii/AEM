\subsection{Special Matrices}

\subsubsection*{Zero Matrix}

\begin{conceptbox}
    A \textbf{zero matrix} is a matrix whose entries are all zero.
\end{conceptbox}

\[
    O =
    \begin{bmatrix}
        0 & 0 \\
        0 & 0
    \end{bmatrix}
\]

\subsubsection*{Identity Matrix}

\begin{conceptbox}
    The \textbf{identity matrix} $I_n$ is a square matrix with 1's on the main diagonal and 0's elsewhere.
\end{conceptbox}

\[
    I_3 =
    \begin{bmatrix}
        1 & 0 & 0 \\
        0 & 1 & 0 \\
        0 & 0 & 1
    \end{bmatrix}
\]

It satisfies:

\[
    AI = IA = A
\]

\subsubsection*{Symmetric Matrix}

\begin{conceptbox}
    A matrix $A$ is \textbf{symmetric} if
    \[
        A^T = A.
    \]
\end{conceptbox}

\begin{examplebox}
    \[
        A =
        \begin{bmatrix}
            2 & 3 \\
            3 & 5
        \end{bmatrix}
    \]

    Since $a_{12} = a_{21}$, the matrix is symmetric.
\end{examplebox}
