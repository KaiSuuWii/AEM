\subsection{Matrix Addition and Scalar Multiplication}

\subsubsection*{Matrix Addition}

\begin{conceptbox}
    Two matrices can be added if and only if they have the same size.
\end{conceptbox}

If $A = [a_{ij}]$ and $B = [b_{ij}]$, then

\[
    A + B = [a_{ij} + b_{ij}]
\]

\begin{examplebox}
    \[
        A =
        \begin{bmatrix}
            1 & 2 \\
            3 & 4
        \end{bmatrix},
        \quad
        B =
        \begin{bmatrix}
            5 & -1 \\
            2 & 0
        \end{bmatrix}
    \]

    \[
        A + B =
        \begin{bmatrix}
            6 & 1 \\
            5 & 4
        \end{bmatrix}
    \]
\end{examplebox}

\subsubsection*{Scalar Multiplication}

\begin{conceptbox}
    If $c$ is a scalar and $A = [a_{ij}]$, then
    \[
        cA = [ca_{ij}]
    \]
\end{conceptbox}

\begin{examplebox}
    \[
        2
        \begin{bmatrix}
            1 & -3 \\
            4 & 2
        \end{bmatrix}
        =
        \begin{bmatrix}
            2 & -6 \\
            8 & 4
        \end{bmatrix}
    \]
\end{examplebox}
