\section*{Quiz: Complex Numbers and Complex Functions}

\textbf{Instructions.} Answer all items completely. Show all necessary solutions, including algebraic simplification, branch choices, and justification of analyticity where needed.

\begin{enumerate}
    \item \textbf{Polar Form, Powers, and Roots}

    Let
    \[
        z=-8+8\sqrt{3}\,i.
    \]
    \begin{enumerate}
        \item Express $z$ in polar form $z=re^{i\theta}$ using the principal argument.
        \item Find all complex numbers $w$ such that
        \[
            w^4=z.
        \]
        \item Plot or describe geometrically the location of the four roots in the complex plane.
    \end{enumerate}

    \item \textbf{Complex Mapping}

    Consider the mapping
    \[
        w=f(z)=\frac{1}{z-i}.
    \]
    Let $C$ be the horizontal line
    \[
        z=x+2i, \qquad -\infty < x < \infty.
    \]
    Find the image of $C$ in the $w$-plane and describe its geometry completely.

    \item \textbf{Analyticity and the Cauchy--Riemann Equations}

    Let
    \[
        f(z)=u(x,y)+iv(x,y),
    \]
    where
    \[
        u(x,y)=x^3-3xy^2+2x,
        \qquad
        v(x,y)=3x^2y-y^3+ay.
    \]
    Determine the value of the real constant $a$ such that $f$ is analytic everywhere. Then express $f(z)$ purely in terms of $z$.

    \item \textbf{Complex Integration}

    Evaluate
    \[
        \int_C \overline{z}\,dz
    \]
    along each of the following paths from $0$ to $1+i$:
    \begin{enumerate}
        \item $C_1$: the line segment $z(t)=(1+i)t$, $0\le t\le 1$;
        \item $C_2$: the polygonal path from $0$ to $1$, then from $1$ to $1+i$.
    \end{enumerate}
    Based on your answers, explain whether $\overline{z}$ has path-independent integrals.

    \item \textbf{Complex Exponential and Logarithm}

    Solve
    \[
        e^z=-2+2i
    \]
    for all complex numbers $z$. Then identify the principal solution using the principal argument.
\end{enumerate}
